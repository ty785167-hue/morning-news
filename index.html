<script>

    // 設定

    const WEATHER_URL = "https://api.open-meteo.com/v1/forecast?latitude=35.6895&longitude=139.6917&current=temperature_2m,weather_code";

    const NEWS_SOURCES = [

        { name: "REUTERS", url: "https://www.reutersagency.com/feed/?best-topics=political-general&post_type=best" },

        { name: "CNN", url: "http://rss.cnn.com/rss/edition_world.rss" },

        { name: "BBC", url: "https://feeds.bbci.co.uk/news/world/rss.xml" }

    ];


    // 天気コードを日本語に変換

    function getWeatherDesc(code) {

        const mapping = { 0: "晴天", 1: "ほぼ晴れ", 2: "薄曇り", 3: "曇り", 45: "霧", 51: "霧雨", 61: "小雨", 71: "小雪", 95: "雷雨" };

        return mapping[code] || "不明";

    }


    async function updateWeather() {

        try {

            const res = await fetch(WEATHER_URL);

            const data = await res.json();

            document.getElementById('weather').innerHTML = `

                <div class="temp">${Math.round(data.current.temperature_2m)}°C</div>

                <div class="condition">Tokyo / ${getWeatherDesc(data.current.weather_code)}</div>

            `;

        } catch (e) { console.error("Weather error"); }

    }


    async function updateNews() {

        const feed = document.getElementById('news-feed');

        let allNews = [];


        for (const source of NEWS_SOURCES) {

            try {

                const res = await fetch(`https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent(source.url)}`);

                const data = await res.json();

                if (data.status === 'ok') {

                    // 各ソースから上位3件を抽出

                    const items = data.items.slice(0, 3).map(item => ({

                        source: source.name,

                        title: item.title,

                        link: item.link,

                        time: new Date(item.pubDate)

                    }));

                    allNews = [...allNews, ...items];

                }

            } catch (e) { console.error(source.name + " error"); }

        }


        // 時間順に並び替え

        allNews.sort((a, b) => b.time - a.time);


        feed.innerHTML = allNews.map(item => `

            <a href="${item.link}" target="_blank" class="news-item">

                <div class="news-source">${item.source}</div>

                <p class="news-headline">${item.title}</p>

                <div class="news-time">${item.time.toLocaleTimeString('ja-JP', {hour: '2-digit', minute:'2-digit'})}</div>

            </a>

        `).join('');


        document.getElementById('last-update').innerText = "Last Update: " + new Date().toLocaleTimeString();

    }


    function refresh() {

        updateWeather();

        updateNews();

    }


    // 初回実行

    refresh();


    // 1時間ごとに自動更新 (3600000ミリ秒)

    setInterval(refresh, 3600000);


    // ページに戻ってきた時（タブを切り替えた時など）に更新

    window.addEventListener('focus', refresh);

</script>
